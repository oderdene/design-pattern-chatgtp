# 8. **Observer Design Pattern-–∏–π–≥ –ø—Ä–æ–≥—Ä–∞–º—á–ª–∞–ª—ã–Ω —Ö—ç–ª –¥—ç—ç—Ä —Ö—ç—Ä—Ö—ç–Ω —Ö—ç—Ä—ç–≥–∂“Ø“Ø–ª–¥—ç–≥ –≤—ç?**

```python
from abc import ABC, abstractmethod

# Subject –∏–Ω—Ç–µ—Ä—Ñ—ç–π—Å
class Subject(ABC):
    def __init__(self):
        self._observers = []

    def attach(self, observer):
        self._observers.append(observer)

    def detach(self, observer):
        self._observers.remove(observer)

    def notify(self):
        for observer in self._observers:
            observer.update(self)

# Observer –∏–Ω—Ç–µ—Ä—Ñ—ç–π—Å
class Observer(ABC):
    @abstractmethod
    def update(self, subject):
        pass

# Concrete Subject
class YouTubeChannel(Subject):
    def __init__(self, name):
        super().__init__()
        self.name = name
        self.videos = []

    def upload_video(self, title):
        self.videos.append(title)
        print(f"[{self.name}] New video uploaded: {title}")
        self.notify()

# Concrete Observer
class Subscriber(Observer):
    def __init__(self, username):
        self.username = username

    def update(self, subject):
        print(f"{self.username} received notification: New video on {subject.name} - {subject.videos[-1]}")

# –ê—à–∏–≥–ª–∞—Ö –Ω—å

```
